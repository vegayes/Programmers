    class Solution {
        public String solution(String[] cards1, String[] cards2, String[] goal) {
            int saveIndex1 = 0;
            int saveIndex2 = 0;

            for (int i = 0; i < goal.length; i++) {
                if (saveIndex1 < cards1.length && cards1[saveIndex1].equals(goal[i])) {
                    saveIndex1++;
                } else if (saveIndex2 < cards2.length && cards2[saveIndex2].equals(goal[i])) {
                    saveIndex2++;
                } else {
                    // 현재 goal[i]를 만들 수 없음
                    return "No";
                }
            }

            // 모든 단어를 순서대로 확인한 후에도 goal을 만들지 못한 경우
            if (saveIndex1 == cards1.length && saveIndex2 == cards2.length) {
                return "Yes";
            } else {
                return "No";
            }
        }
    }

class Solution_1 {
    public int solution(int num) {
        int answer = 0;

        while(num != 1){
            if(num % 2 == 0){ // 1) 입력된 수가 짝수라면 2로 나눔
                num = num/2;
            }else { // 2) 홀수라면 3을 곱하고 1을 더함.
                num = (num*3) +1;
            }
            answer++;

            if(answer >= 500){
                answer = -1;
                break;
            }
        }
        return answer;
    }
}
